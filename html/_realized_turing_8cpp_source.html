205void OneStepTur(bool& error)
206{
207 bool flag = 1;
208 for (int i = 0; i < MyTuring.UserVec.size() && flag; ++i)
209 {
210 struct data temp = MyTuring.UserVec[i];
211 if (temp.name == MyTuring.head && temp.symbol == MyTuring.ribbon[MyTuring.position])
212 {
213 MyTuring.ribbon[MyTuring.position] = temp.symbolAdd;
214 OneAction(temp,error);
215 if (error) break;
216 MyTuring.head = temp.next;
217 flag = 0;
218 }
219 else if (!MyTuring.alphabet.count(temp.symbol))
220 {
221 PrintErrorAlpha();
222 error = 1;
223 break;
224 }
225 }
